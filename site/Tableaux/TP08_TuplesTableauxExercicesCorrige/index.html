    def moyenneCoeff(tab):

    sommecoeffs=0
    nb2notes=len(tab)

    return moy

    from random import choice

def cadavre_exquis(noms, verbes, complements):
    nom = noms
    verbe = verbes
    complement = complements
    phrase = choice(nom)+choice(verbe)+choice(complement)
    return phrase
